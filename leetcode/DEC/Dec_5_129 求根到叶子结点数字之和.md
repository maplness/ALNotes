# Dec_5_129 求根到叶子结点数字之和
``` c++
class Solution {
public:
    int dfs(TreeNode *node, int prevSum){
        if(!node) return 0;
        int sum = prevSum * 10 + node->val;
        if(!node->left && !node->right){
            return sum;
        }
        else return dfs(node->left, sum) + dfs(node->right, sum);
    }

    int sumNumbers(TreeNode* root) {
        int sum = 0;
        
        return dfs(root,  sum);
    }
};
```
