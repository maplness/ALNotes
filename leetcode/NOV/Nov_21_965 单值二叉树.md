Nov_21_965 单值二叉树

``` C++
class Solution {
public:
	bool isUnivalTree(TreeNode* root) {
		if(!root) return true;
		stack<TreeNode *> s;
		int p = root->val;
		while(!s.empty() || root){
			if(root){
				if (p != root->val) {
					return false;
				}
				s.push(root);
				root = root->left;
			}else {
				root = s.top();
				root = root->right;
				s.pop();
			}
			
		}
		return true;
	}
};
```
