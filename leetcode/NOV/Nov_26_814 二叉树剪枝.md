# Nov_26_814 二叉树剪枝

``` c++
class Solution {
public:
    TreeNode* pruneTree(TreeNode* root) {
        return conatinsOne(root) ? root : nullptr;
    }

    bool conatinsOne(TreeNode* root)
    {
        if(!root) return false;
        bool left = conatinsOne(root->left);
        bool right = conatinsOne(root->right);
        if(!left) root->left = nullptr;
        if(!right) root->right = nullptr;
        return root->val == 1 || left || right; 
    }
};
```
